```python linenums="1"
import numpy as np
import matplotlib.pyplot as plt
```
画图
=== "代码"
    ```python linenums="1"
    #绘图数据
    x = np.arange(0, 2*np.pi, 0.7)
    y1 = np.sin(x)
    y2 = np.cos(x)
    #绘直线
    plt.plot(x, y1)
    plt.plot(x, y2)
    #绘散点图
    plt.scatter(x, y1)
    plt.scatter(x, y2)
    #或者也可以
    #plt.plot(x, y1,'.-')

    plt.show()#显示图片
    ```
