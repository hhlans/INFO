f2w = fw*xf2w;

xf1 = f1/f;
xf2 = f2/f;

%>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>物料衡算

d1w = dw*xd1w;
d2w = dw-d1w;
d1 = d1w/M1;
d2 = d2w/M2;

xd1 = d1/d;
xd2 = d2/d;
ww = fw-dw;
w1w = ww*xw1w;
w2w = ww-w1w;
w1 = w1w/M1;
w2 = w2w/M2;

xw1 = w1/w;
xw2 = w2/w;

%>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>antoine
a1 = 20.7936;
b1 = 2788.51;
c1 = 52.36;
a2 = 20.9065;
b2 = 3096.52;
c2 = 53.67;

ps2 = @(ts)exp(a2-b2./(ts-c2));
ts1 = @(ps)b1./(a1-log(ps))+c1;
ts2 = @(ps)b2./(a2-log(ps))+c2;
x1_t = @(ts)(p-ps2(ts))./(ps1(ts)-ps2(ts));
y1_t = @(ts)ps1(ts)./p.*x1_t(ts);

%>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>平衡线
apa = @(ts)ps1(ts)./ps2(ts);%alpha(T)
y_ = @(x_, a)a*x_./(1+a*x_-x_);%y(x)
x_ = @(y, a)y./(a-a*y+y);%x(y)
bp(xd1, 1-xd1, p);
%塔顶塔底温度
td = dp(xd1, xd2, p);

%挥发度

alphaw = apa(tw);

%最小理论板数 最小回流比
ddtheta = @(theta)(alpha_*xf1)/(alpha_-theta)+(xf2)/(1-theta)-1+q;
thret = fzero(ddtheta, 1.4);

ye2 = 1-ye1;

td_ = (tfb+td)/2;

Mqd = M1*x11+M2*x12;

louaf = pf*Maf/Rc/tfb/1e3;%进料板

louad_ = (louad+louaf)/2;%精馏段

E_ = @(lh, E_)lh/lw^2.5-2147*((sqrt(1-(lw/D)^2/(E_^3))-sqrt(1-(lw/D)^2))/(lw/D)/E_)^(3/2);

%受液盘

n = 1158e3/t/t*Aa/1e6;

function t = bp(x1, x2, p)
    global ps1 ps2 ts1 ts2;

function t = bp(x1, x2, p)
    global ps1 ps2 ts1 ts2;
    t = 300;
    delta = 0.1;
    while delta>0.0001
        k1 = ps1(t)/p;
        k2 = ps2(t)/p;
        delta = k1*x1+k2*x2;
        if(x1>0.5)
            k1 = k1/delta;
            p1_ = k1*p;
            t = ts1(p1_);
            k2 = ps2(t)/p;
        else
            k2 = k2/delta;
            p2_ = k2*p;
            t = ts2(p2_);
            k1 = ps1(t)/p;
        end
        delta = 1-delta;
    end
end
